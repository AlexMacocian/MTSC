27  int length = int.Parse(request[HttpMessage.EntityHeadersEnum.ContentLength]);
28  int step = 0, parsedLength = 0;
29  StringBuilder fieldBuilder = new StringBuilder();
30  StringBuilder valueBuilder = new StringBuilder();
31  /*
32  * Parse request body to obtain the form data.
33  * Example of a form: field1=value1&field2=value2
34  */
35  for (int i = 0; i < request.Body.Length; i++)
36  {
37  if(request.Body[i] == '\n')
38  {
39  continue;
40  }
41  else if(step == 0)
42  {
43  parsedLength++;
44  if(request.Body[i] == '=')
45  {
46  step = 1;
47  }
48  else
49  {
50  fieldBuilder.Append((char)request.Body[i]);
51  }
52  }
53  else
54  {
55  parsedLength++;
56  if(request.Body[i] == '&')
57  {
58  step = 0;
59  formData[fieldBuilder.ToString()] = valueBuilder.ToString();
60  fieldBuilder.Clear();
61  valueBuilder.Clear();
62  }
63  else
64  {
65  valueBuilder.Append((char)request.Body[i]);
66  if(parsedLength == length)
67  {
68  formData[fieldBuilder.ToString()] = valueBuilder.ToString();
69  }
70  }
71  }
72  }

74  }
